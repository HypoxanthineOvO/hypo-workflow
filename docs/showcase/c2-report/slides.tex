\section{动机与问题}
\WorkflowSectionFrame{Part I}{为什么需要一个新的工程层}{assets/gpt-image/tool-evolution-diary.png}

\begin{frame}{个人 AI 工具迁移路径}
  \begin{columns}[T,totalwidth=\textwidth]
    \column{0.58\textwidth}
    \begin{block}{使用路径}
      \begin{enumerate}
        \item Web GPT / Gemini / Cherry Studio + DeepSeek
        \item GitHub Copilot
        \item AntiGravity
        \item Claude Code
        \item Skills / Superpowers
        \item Notion AI Opus + Codex
        \item Hypo-Workflow
      \end{enumerate}
    \end{block}
    \column{0.38\textwidth}
    \begin{alertblock}{关键观察}
      每一次工具跃迁都在修补上一个阶段暴露的结构性问题，而不是单纯追求“更会写代码”。
    \end{alertblock}
  \end{columns}
\end{frame}

\begin{frame}{真实经历不是背景，而是需求来源}
  \small
  \begin{tabularx}{\textwidth}{p{0.18\linewidth}p{0.24\linewidth}X}
    \toprule
    项目 & 工具组合 & 暴露的问题 \\
    \midrule
    Hypo-LaTeX & Copilot + 手工 prompt & 写 CHANGELOG、Git 规范、release notes 都要反复喂 Prompt；Copilot 连续被封 3 个账号后，单一工具入口不再可信。 \\
    Bill & Claude & 前期搭建快，后期 UI 显示问题反复修、越改越乱，缺少文档和视觉验收证据。 \\
    Agent & Claude + Superpowers & Superpowers 作为早期 Harness 前期很有效，但持续修 Bug 时缺少全局观，局部修改持续引入新问题。 \\
    Research / Info & Notion AI + Codex & 规划、文档、执行分裂，作者变成 Prompt 搬运工。 \\
    \bottomrule
  \end{tabularx}
\end{frame}

\begin{frame}{我真正踩到的坑}
  \begin{columns}[T,totalwidth=\textwidth]
    \column{0.32\textwidth}
    \begin{block}{网页模型阶段}
      \begin{itemize}
        \item 不能读完整 repo
        \item 只能靠人描述代码
        \item 重复 prompt 成本极高
      \end{itemize}
    \end{block}
    \column{0.32\textwidth}
    \begin{block}{Copilot / AntiGravity}
      \begin{itemize}
        \item Copilot 补全强但规划弱
        \item AntiGravity 自动化感强但不可控
        \item 验证闭环都不够硬
      \end{itemize}
    \end{block}
    \column{0.32\textwidth}
    \begin{block}{Claude Code 之后}
      \begin{itemize}
        \item 单次对话能力很强
        \item 但行为约束仍要重复粘贴
        \item 长期历史仍会逐渐丢失
      \end{itemize}
    \end{block}
  \end{columns}
\end{frame}

\begin{frame}{为什么“多写一点 Prompt”仍然不够}
  \begin{block}{核心判断}
    Prompt 可以补救一次，但不能稳定替代状态系统、计划系统和验证系统。
  \end{block}
  \begin{itemize}
    \item 上下文窗口天然是局部的
    \item 规则越依赖人工复述，越容易丢
    \item 模型会在错误前提上持续自我强化
    \item 中断之后很难恢复到同一工程状态
  \end{itemize}
\end{frame}

\begin{frame}{AI-assisted coding 的六个核心约束}
  \begin{columns}[T,totalwidth=\textwidth]
    \column{0.48\textwidth}
    \begin{block}{Context / Prompt}
      \begin{itemize}
        \item context management
        \item repeated prompt feeding
        \item self-deception risk
      \end{itemize}
    \end{block}
    \column{0.48\textwidth}
    \begin{block}{Process / Recovery}
      \begin{itemize}
        \item style persistence
        \item iteration drift
        \item interruption / recovery
      \end{itemize}
    \end{block}
  \end{columns}
  \vspace{0.3cm}
  \begin{alertblock}{结论}
    主问题不是“能不能生成代码”，而是“能不能持续维持工程约束”。
  \end{alertblock}
\end{frame}

\section{Harness Engineering}
\begin{frame}{Harness Engineering}
  \begin{definition}{定义}{def:harness}
    Harness Engineering 是围绕 AI Agent 构建外部工程约束层：把 context、plans、state、rules、evaluation、recovery 与 collaboration protocol 固化到文件和工具里。
  \end{definition}
  \vspace{0.2cm}
  \begin{itemize}
    \item execution loop
    \item layering
    \item behavior solidification
    \item automatic maintenance
  \end{itemize}
\end{frame}

      \end{itemize}

        \item 平台 adapter 可以只做投影
        \item 规划、执行、归档都可审计
      \end{itemize}
    \end{block}
  \end{columns}
\end{frame}

\begin{frame}{README 自动更新 + Skill 质量治理}

        \item SKILL 结构和命名规范
        \item \texttt{skill-quality} rule
        \item Codex / Claude Code / OpenCode 映射
      \end{itemize}
    \end{block}
  \end{columns}
\end{frame}

\begin{frame}{OpenCode 状态面板：为什么是只读投影}
  \begin{columns}[T,totalwidth=\textwidth]
    \column{0.56\textwidth}
    \includegraphics[width=\linewidth]{assets/opencode-observability.png}
    \column{0.40\textwidth}
    \begin{block}{Sidebar}
      Cycle、Feature Queue、当前 Milestone、最近事件
    \end{block}
    \begin{block}{Footer}
      当前 step、进度、score、heartbeat、duration / token / cost
    \end{block}
    \begin{alertblock}{判断}
      状态面板不应该成为第二套状态系统，它只应投影 \texttt{.pipeline/}。
    \end{alertblock}
  \end{columns}
\end{frame}

\begin{frame}{Batch Planning：加一个中层调度器}
  \begin{itemize}
    \item \texttt{/hw:plan --batch}：一次 Discover 覆盖多个 Feature
    \item upfront / just-in-time decomposition
    \item \texttt{/hw:plan --insert}：自然语言 queue edit
    \item auto-chain + \texttt{gate: confirm}
    \item Feature Queue 与 metrics 分离
  \end{itemize}
  \vspace{0.35cm}
  \begin{block}{核心价值}
    把“每个功能重新走一遍完整计划流程”压缩成“先统一 Discover，再按顺序推进执行”。
  \end{block}
\end{frame}

\begin{frame}{Chat Mode：把轻量追加讨论纳入正式轨道}
  \begin{itemize}
    \item \texttt{/hw:chat} 重载 workflow context
    \item \texttt{chat:} state + \texttt{chat\_entry} log
    \item SessionStart / Stop Hook 负责恢复与 summary fallback
    \item 规模超限时提示升级到 Patch
  \end{itemize}
  \vspace{0.35cm}
  \begin{alertblock}{边界感}
    Chat Mode 不是“再开一个随便聊的窗口”，而是主线外的轻量、可恢复、可落盘会话轨道。
  \end{alertblock}
\end{frame}
